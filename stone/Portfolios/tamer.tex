%% employing the CTeX for better localization
%% the default fontset `fandol` has trouble
%%   in displaying rarely used characters
\usepackage[heading = true, fontset = macnew]{ctex}

\setcounter{tocdepth}{1}
\setcounter{secnumdepth}{3}

%% reset chapter counters for parts
\usepackage{chngcntr}
\counterwithin{chapter}{part}

%% set chapter format
\ctexset{
  chapter/name = {},
  chapter/number = \arabic{chapter},
  section/number = \thesection,
  section/format = \Large\bfseries
}
